cycle_to_suffix <- function(cycle) {
  map <- c(
    "1999-2000" = "A", "2001-2002" = "B", "2003-2004" = "C", "2005-2006" = "D",
    "2007-2008" = "E", "2009-2010" = "F", "2011-2012" = "G", "2013-2014" = "H",
    "2015-2016" = "I", "2017-2018" = "J", "2019-2020" = "K"
  )
  out <- unname(map[cycle])
  if (is.na(out)) stop("Cycle not mapped: ", cycle)
  out
}

safe_nhanes <- function(table_name, verbose = TRUE) {
  out <- tryCatch(nhanesA::nhanes(table_name), error = function(e) NULL)
  if (is.null(out) && verbose) message("Unavailable table: ", table_name)
  out
}

resolve_table <- function(cycle, candidates, domain = NULL, verbose = TRUE) {
  suf <- cycle_to_suffix(cycle)
  to_try <- paste0(candidates, "_", suf)
  for (tb in to_try) {
    dat <- safe_nhanes(tb, verbose = FALSE)
    if (!is.null(dat)) {
      if (verbose) message("Using table ", tb, " for cycle ", cycle)
      return(list(code = candidates[1], table = tb, data = dat))
    }
  }
  NULL
}

coalesce_num <- function(df, vars) {
  vars <- vars[vars %in% names(df)]
  if (length(vars) == 0) return(rep(NA_real_, nrow(df)))
  v <- lapply(df[vars], function(x) suppressWarnings(as.numeric(as.character(x))))
  dplyr::coalesce(!!!v)
}

parse_yes_no <- function(x) {
  x_num <- suppressWarnings(as.numeric(as.character(x)))
  x_chr <- toupper(trimws(as.character(x)))
  case_when(
    x_num == 1 ~ 1,
    x_num == 2 ~ 0,
    str_detect(x_chr, "^YES$") ~ 1,
    str_detect(x_chr, "^NO$") ~ 0,
    TRUE ~ NA_real_
  )
}

}

compute_egfr_2021 <- function(scr_mgdl, age, sex_raw) {
  sex_chr <- toupper(trimws(as.character(sex_raw)))
  sex_num <- suppressWarnings(as.numeric(as.character(sex_raw)))
  male <- case_when(
    sex_num == 1 ~ TRUE,
    sex_num == 2 ~ FALSE,
    sex_chr == "MALE" ~ TRUE,
    sex_chr == "FEMALE" ~ FALSE,
    TRUE ~ NA
  )
  kappa <- ifelse(male, 0.9, 0.7)
  alpha <- ifelse(male, -0.302, -0.241)
  sex_mult <- ifelse(male, 1, 1.012)
  142 * pmin(scr_mgdl / kappa, 1)^alpha * pmax(scr_mgdl / kappa, 1)^(-1.200) * (0.9938^age) * sex_mult
}

compute_acr <- function(ua, ucr, acr_direct = NULL) {
  calc <- ifelse(!is.na(ua) & !is.na(ucr) & ucr > 0, (ua / ucr) * 100, NA_real_)
  dplyr::coalesce(acr_direct, calc)
}

compute_phenoage_components <- function(df, include_scr = TRUE, include_alb = TRUE) {
  # Levine et al. (2018) coefficients expect: albumin g/L, creatinine µmol/L, glucose mmol/L.
  # NHANES provides: albumin g/dL, creatinine mg/dL, glucose often mg/dL (we standardize to mg/dL in pipeline).
  alb_gL   <- df$alb * 10
  scr_umol <- df$scr * 88.4
  glu_mmol <- df$glu / 18.018
  alb_term <- if (include_alb) (-0.0336 * alb_gL) else 0
  scr_term <- if (include_scr) ( 0.0095 * scr_umol) else 0
  glu_term <- 0.1953 * glu_mmol

  xb <- -19.907 +
    alb_term +
    scr_term +
    glu_term +
    0.0954 * log(pmax(df$crp, 1e-6)) -
    0.0120 * df$lym +
    0.0268 * df$mcv +
    0.3306 * df$rdw +
    0.00188 * df$alp +
    0.0554 * df$wbc +
    0.0804 * df$age

  xb <- pmin(pmax(xb, -50), 50)
  gamma_const <- (exp(120 * 0.0076927) - 1) / 0.0076927
  hazard <- exp(xb) * gamma_const
  mort <- -expm1(-hazard)
  mort <- pmin(pmax(mort, 1e-12), 1 - 1e-12)
  pheno <- 141.50 + log(-0.00553 * log1p(-mort)) / 0.090165

  list(xb = xb, mort = mort, phenoage = pheno)
}

safe_phenoage <- function(df) {
  compute_phenoage_components(df, include_scr = TRUE, include_alb = TRUE)$phenoage
}

safe_qgcomp_extract <- function(fit_obj, model_label, n_obs) {
  if (is.null(fit_obj)) {
    return(tibble(model = model_label, beta = NA_real_, conf.low = NA_real_, conf.high = NA_real_, p.value = NA_real_, N = n_obs))
  }
  cf <- coef(fit_obj)
  psi_name <- grep("^psi", names(cf), value = TRUE)[1]
  if (is.na(psi_name)) psi_name <- grep("psi", names(cf), value = TRUE)[1]
  if (is.na(psi_name)) {
    message("qgcomp terms for [", model_label, "]: no psi found in ", paste(names(cf), collapse = ", "))
    return(tibble(model = model_label, beta = NA_real_, conf.low = NA_real_, conf.high = NA_real_, p.value = NA_real_, N = n_obs))
  }
  beta <- unname(cf[psi_name])
  vc <- tryCatch(vcov(fit_obj), error = function(e) NULL)
  if (!is.null(vc) && psi_name %in% rownames(vc) && psi_name %in% colnames(vc)) {
    se <- sqrt(vc[psi_name, psi_name])
    lo <- beta - 1.96 * se
    hi <- beta + 1.96 * se
    p_val <- 2 * pnorm(-abs(beta / se))
  } else {
    lo <- hi <- p_val <- NA_real_
  }
  tibble(model = model_label, beta = beta, conf.low = lo, conf.high = hi, p.value = p_val, N = n_obs)
}

resolve_weight_var <- function(df, weight_type = "auto", exposure_family = "pfas") {
  cols <- names(df)
  if (weight_type != "auto" && weight_type != "MEC") {
    if (!(weight_type %in% cols)) stop("Requested weight variable not found: ", weight_type)

In PFAS analyses, final design should normally use the PFAS-specific subsample weight identified by the audit/auto-selection logic."))
  stop("No supported NHANES weight variable found.")
}

apply_lod_rule <- function(df, varname) {
  x <- suppressWarnings(as.numeric(as.character(df[[varname]])))

  base <- sub("^(LBX|URX)", "", varname)
  lc_candidates <- c(
    paste0(varname, "LC"),
    paste0("LBD", base, "LC"),
    paste0("URD", base, "LC"),
    # NHANES often uses *L (not *LC) as the detection-limit flag.
    paste0(varname, "L"),
    paste0("LBD", base, "L"),
    paste0("URD", base, "L")
  )
  lc_candidates <- lc_candidates[lc_candidates %in% names(df)]

  if (length(lc_candidates) > 0) {
    lc <- df[[lc_candidates[1]]]
    lc_num <- suppressWarnings(as.numeric(as.character(lc)))
    lc_chr <- toupper(trimws(as.character(lc)))
    below <- (!is.na(lc_num) & lc_num == 1) | str_detect(lc_chr, "BELOW|<")
    below[is.na(below)] <- FALSE
    # NHANES convention: when LC==1, the concentration column typically holds the LOD value.
    # Standard replacement: value / sqrt(2).
    x[below] <- x[below] / sqrt(2)
    # If value is still NA/non-positive after that, use smallest observed LOD / sqrt(2).
    lod_vals <- x[below & x > 0]
    fallback <- if (length(lod_vals) > 0) min(lod_vals, na.rm = TRUE) else NA_real_
    x[below & (is.na(x) | x <= 0)] <- fallback
    return(x)
  }
  # If no LOD indicator exists, do not impute missing values.
  x
}

download_cycle <- function(cycle, exposure_family = "pfas", verbose = TRUE) {
  core_domains <- c("DEMO", "BMX", "SMQ", "ALQ", "PAQ", "DIQ", "BPQ", "MCQ", "BIOPRO", "CBC", "CRP", "ALB_CR", "GLU")
  exp_domains <- switch(
    exposure_family,
    pfas = c("PFAS"),
    metals = c("BLOOD_METALS", "URINE_METALS"),
    both = c("PFAS", "BLOOD_METALS", "URINE_METALS"),
    c("PFAS")
  )
  all_domains <- c(core_domains, exp_domains)

  resolved <- lapply(all_domains, function(dom) resolve_table(cycle, table_candidates[[dom]], domain = dom, verbose = verbose))
  names(resolved) <- all_domains
  resolved <- resolved[!vapply(resolved, is.null, logical(1))]
  if (!"DEMO" %in% names(resolved)) return(NULL)

  dat <- resolved$DEMO$data
  if (!"SEQN" %in% names(dat)) return(NULL)

  for (nm in setdiff(names(resolved), "DEMO")) {
    d <- resolved[[nm]]$data
    # Rename glucose vars from GLU table before joining so they do not collide with BIOPRO/L40.
    if (nm == "GLU") {
      if ("LBXGLU" %in% names(d)) names(d)[names(d) == "LBXGLU"] <- "LBXGLU_GLU"
      if ("LBXSGL" %in% names(d)) names(d)[names(d) == "LBXSGL"] <- "LBXSGL_GLU"
    }
    if ("SEQN" %in% names(d)) dat <- left_join(dat, d, by = "SEQN")
  }
  # After merge: define explicit glucose sources. Primary analysis uses MEC chemistry glucose for measurement consistency in PhenoAge construction.
  dat$glu_mec  <- coalesce_num(dat, c("LBXGLU", "LBXSGL"))
  dat$glu_glu  <- coalesce_num(dat, c("LBXGLU_GLU", "LBXSGL_GLU"))
  dat$glu     <- dat$glu_mec
  dat$cycle <- cycle
  # Store provenance for key tables so we can audit weight/source compatibility.
  if ("CRP" %in% names(resolved)) dat$crp_table <- resolved$CRP$table
  if ("GLU" %in% names(resolved)) dat$glu_table <- resolved$GLU$table
  # Warn if joins created .x/.y duplicate columns (would break coalesce_num).
  dup_suffix_cols <- names(dat)[grepl("\\.x$|\\.y$", names(dat))]
  if (length(dup_suffix_cols) > 0) {
    warning("Cycle ", cycle, ": duplicate columns from join: ", paste(dup_suffix_cols, collapse = ", "))
  }
  dat
}

check_raw_levels <- function(df, var, allowed_patterns) {
  if (!(var %in% names(df))) return(tibble(variable = var, cycle = NA_character_, n_unexpected = NA_integer_, unexpected_examples = NA_character_))

    mutate(.raw = toupper(trimws(as.character(.data[[var]])))) %>%
    group_by(cycle) %>%
    summarise(

      n_unexpected = sum(!is.na(.raw) & .raw != "" & !str_detect(.raw, allowed_patterns)),
      unexpected_examples = paste(head(unique(.raw[!is.na(.raw) & .raw != "" & !str_detect(.raw, allowed_patterns)]), 6), collapse = ", "),
      .groups = "drop"
    ) %>%
    mutate(variable = var) %>%
    select(variable, cycle, n_nonmissing, n_unexpected, unexpected_examples)
}

# --- ISOMER HARMONIZATION (2013-2018) ---
# NHANES split PFOS and PFOA into linear and branched isomers.
# We must apply the LOD/sqrt(2) rule to the individual isomers FIRST, then sum them.

isomer_map <- list(
  pfos_lin = c("SSNPFOS", "LBXNFOS"),
  pfos_br  = c("SSMPFOS", "LBXMFOS"),
  pfoa_lin = c("SSNPFOA", "LBXNFOA"),
  pfoa_br  = c("SSBPFOA", "LBXBFOA")
)

lc_map <- list(
  pfos_lin = c("SDNPFOSL", "LBDNFOSL"),
  pfos_br  = c("SDMPFOSL", "LBDMFOSL"),
  pfoa_lin = c("SDNPFOAL", "LBDNFOAL"),
  pfoa_br  = c("SDBPFOAL", "LBDBFOAL")
)

for (chem in c("pfos", "pfoa")) {
  # Get available variables for the cycle
  lin_vars <- isomer_map[[paste0(chem, "_lin")]]
  br_vars  <- isomer_map[[paste0(chem, "_br")]]
  lin_lc   <- lc_map[[paste0(chem, "_lin")]]
  br_lc    <- lc_map[[paste0(chem, "_br")]]
  
  lin_val <- coalesce_num(dat, lin_vars)
  br_val  <- coalesce_num(dat, br_vars)
  lin_lc_val <- coalesce_num(dat, lin_lc)
  br_lc_val  <- coalesce_num(dat, br_lc)
  
  # Apply LOD rule: if indicator == 1, divide by sqrt(2)
  lin_val <- ifelse(!is.na(lin_lc_val) & lin_lc_val == 1, lin_val / sqrt(2), lin_val)
  br_val  <- ifelse(!is.na(br_lc_val) & br_lc_val == 1, br_val / sqrt(2), br_val)
  
  # Sum them to recreate the total concentration
  summed_val <- lin_val + br_val
  
  # Inject back into the legacy total variables (LBXPFOS or LBXPFOA)
  target_var <- ifelse(chem == "pfos", "LBXPFOS", "LBXPFOA")
  
  if (target_var %in% names(dat)) {
    dat[[target_var]] <- coalesce_num(dat, c(target_var))
    dat[[target_var]] <- ifelse(is.na(dat[[target_var]]) & !is.na(summed_val), summed_val, dat[[target_var]])
  } else {
    dat[[target_var]] <- summed_val
  }
}

weighted_quantile <- function(x, w, probs) {
  keep <- is.finite(x) & is.finite(w) & w > 0
  x <- x[keep]; w <- w[keep]
  if (length(x) == 0) return(rep(NA_real_, length(probs)))
  ord <- order(x)
  x <- x[ord]; w <- w[ord]
  cw <- cumsum(w) / sum(w)
  sapply(probs, function(p) {
    idx <- which(cw >= p)[1]
    if (is.na(idx)) x[length(x)] else x[idx]
  })
}

smd_cont <- function(o0, o1) {
  if (any(!is.finite(c(o0$est, o1$est, o0$var, o1$var)))) return(NA_real_)
  denom <- sqrt((o0$var + o1$var) / 2)
  if (!is.finite(denom) || denom <= 0) return(NA_real_)
  abs(o1$est - o0$est) / denom
}

  if (any(!is.finite(c(o0$est, o1$est)))) return(NA_real_)
  denom <- sqrt((o0$est * (1 - o0$est) + o1$est * (1 - o1$est)) / 2)
  if (!is.finite(denom) || denom <= 0) return(NA_real_)
  abs(o1$est - o0$est) / denom
}

make_rhs_formula <- function(outcome, main_terms, covar_rhs) {
  rhs <- paste(main_terms, collapse = " + ")
  if (!is.null(covar_rhs) && nzchar(covar_rhs)) rhs <- paste(rhs, covar_rhs, sep = " + ")
  as.formula(paste(outcome, "~", rhs))
}
fit_svy_with_covar_guard <- function(data_df, outcome, main_terms, covars_terms, label) {
  build_formula <- function(covs) {
    covar_rhs <- if (length(covs) > 0) paste(covs, collapse = " + ") else ""
    make_rhs_formula(outcome, main_terms, covar_rhs)
  }
  covs <- covars_terms
  intended <- covs
  f <- build_formula(covs)
  mf <- NULL
  repeat {
    mf <- tryCatch(model.frame(f, data = data_df, na.action = na.omit), error = function(e) NULL)
    if (is.null(mf) || nrow(mf) == 0) break
    factor_terms <- names(mf)[vapply(mf, is.factor, logical(1))]
    bad_factor_terms <- factor_terms[vapply(mf[factor_terms], function(x) nlevels(droplevels(x)) < 2, logical(1))]
    removable <- intersect(covs, bad_factor_terms)
    if (length(removable) == 0) break
    message("Sensitivity [", label, "]: dropping collapsed factor term(s): ", paste(removable, collapse = ", "))
    covs <- setdiff(covs, removable)
    f <- build_formula(covs)
  }
  if (is.null(mf) || nrow(mf) == 0) {
    return(list(
      fit = NULL, n_model = 0L, intended = intended, realized = covs,
      dropped = paste(setdiff(intended, covs), collapse = ", "),
      error = "No complete-case rows for model frame."
    ))
  }
  data_svy <- as.data.frame(data_df)
  for (v in c("phenoage_accel", "phenoage_nokid", "phenoage_nokid_both", "phenoage_accel_unweighted", "mixture_qscore", "ckd", "ckd_sens", "age", "pir", "smoker", "bmi", "weight_combined", "psu", "strata")) {
    if (v %in% names(data_svy)) data_svy[[v]] <- suppressWarnings(as.numeric(as.character(data_svy[[v]])))
  }
  if ("ckd" %in% names(data_svy)) data_svy$ckd <- as.integer(data_svy$ckd)
  for (v in c("sex", "race_eth", "cycle")) {
    if (v %in% names(data_svy)) data_svy[[v]] <- factor(as.character(data_svy[[v]]))
  }
  data_svy$sex      <- droplevels(factor(data_svy$sex))
  data_svy$race_eth <- droplevels(factor(data_svy$race_eth))
  data_svy$cycle    <- droplevels(factor(data_svy$cycle))
  des <- tryCatch(
    svydesign(id = ~psu, strata = ~strata, weights = ~weight_combined, nest = TRUE, data = data_svy),
    error = function(e) svydesign(id = ~1, strata = NULL, weights = ~weight_combined, data = data_svy)
  )
  err <- NULL
  fit <- tryCatch(
    svyglm(f, design = des, family = gaussian()),
    error = function(e) {
      err <<- conditionMessage(e)
      NULL
    }
  )
  list(
    fit = fit,
    n_model = nrow(mf),
    intended = intended,
    realized = covs,
    dropped = paste(setdiff(intended, covs), collapse = ", "),
    error = err
  )
}

13 Cautions and Limitations

  • Cross-sectional NHANES data do not establish temporality.
  • PhenoAge includes kidney-related biomarkers (creatinine, albumin), creating potential circularity with CKD definitions. Sensitivity analyses recompute PhenoAge excluding (a) creatinine only and (b) both creatinine and albumin; both are non-standard derivatives and interpreted as robustness checks.
  • PhenoAge uses Levine (2018) coefficients with correct units: albumin g/L (NHANES g/dL × 10), creatinine µmol/L (mg/dL × 88.4), glucose mmol/L (mg/dL ÷ 18.018). Model B uses phenoage_accel (age-residualized) and does not adjust for age again. Mixture quantiles are computed within each NHANES cycle to avoid temporal confounding (PFAS trends 2003–2018). Survey design uses full PSU/strata when possible; if the design fails (e.g. singleton PSUs), a weights-only fallback is used and noted.
  • qgcomp point estimates are presented descriptively (pooled sampling weights applied, but PSU/strata not incorporated for variance estimation). Statistical inference relies on the survey-weighted regression models (svyglm) above. Bootstrap variance estimation incorporating replicate weights was not implemented because NHANES does not provide replicate weights for environmental subsamples across all cycles.
  • LOD handling follows NHANES detection-flag conventions when indicator columns are available (*LC or *L): values flagged below LOD (indicator == 1) are replaced with value / sqrt(2). When no LOD indicator is available, values are left unchanged (no fallback NA imputation), which improves transparency but can reduce complete-case sample size.
  • Weight selection: a weight audit table is printed and config$weight_type = "auto" applies the top-coverage eligible weight. For PFAS, auto-selection excludes full-sample interview/MEC weights (WTINT2YR, WTMEC2YR) and prioritizes environmental subsample weights (WTS*2YR), including the combined cross-cycle PFAS weight column.
  • The primary PFAS mixture was defined as a fixed four-chemical set available across all retained cycles (PFOS, PFOA, PFNA, and PFHxS), with complete-case requirement on that set. PFDA was not consistently available across cycles and was therefore excluded from the primary mixture and evaluated only in supplementary analyses. Strict complete-case (qgcomp-compatible PFAS subset) modeling is reported as sensitivity. A sensitivity analysis using an available-exposure (adaptive) mixture can be added to assess robustness when using broader PFAS information.
  • Exposure quantile cutpoints are defined within cycle using cycle-specific NHANES sampling weights.
  • PhenoAgeAccel is computed as the residual from a survey-weighted regression of PhenoAge on chronological age (svyglm), ensuring the construct is population-representative. An unweighted residualization is included as sensitivity.
  • CRP unit detection uses a two-part rule: (1) table-name provenance (HSCRP tables imply mg/L), and (2) a conservative distribution heuristic (median > 1.5 and p75 > 3). If either triggers, CRP is converted from mg/L to mg/dL (/ 10). Cycle-level diagnostic tables are printed. PhenoAge is recomputed after any conversions. Glucose is converted from mmol/L to mg/dL if median < 6.
  • Extreme biomarker values are handled by plausibility filters (WBC > 50, glucose > 600, ALP > 500 set to missing). CRP > 1 mg/dL is retained in primary analyses and evaluated in sensitivity analyses excluding potential acute inflammatory states.
  • Glucose is sourced from MEC biochemistry (glu_mec: BIOPRO/L40) for the primary analysis. The separate GLU table (fasting) is joined with renamed variables (LBXGLU_GLU, LBXSGL_GLU) to avoid duplicate columns; it is retained for diagnostics but not used in the primary model.
  • Marginal effects in the interaction model (Table 3) use svycontrast() with design-based degrees of freedom and t-distribution p-values, maintaining full complex-survey consistency.
  • Because complete-case requirements can collapse factor levels in the interaction model frame, Model B reports both intended and realized covariate sets; unsupported factor terms are dropped with explicit logging.
